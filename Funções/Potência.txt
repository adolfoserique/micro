    int Potencia(int x, int N){

	int sign = 0, p = 1;

	if(N < 0){

		N =-N;

		sign = 1;

	}

	while(N > 0){

		p = x*p;

		N--;

	}

	if(sign == 1)

        return 1/p;

	else

		return p;

}
